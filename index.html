from datetime import timedelta

# Ensure date column is datetime
df_long['Date'] = pd.to_datetime(df_long['Date'])

# Filter and copy absent Fridays
fridays = df_long[(df_long['DayOfWeek'] == 'Friday') & (df_long['Absence_Flag'] == 1)].copy()
fridays['Next_Monday'] = fridays['Date'] + timedelta(days=3)

# Match with absent Mondays
mondays = df_long[(df_long['DayOfWeek'] == 'Monday') & (df_long['Absence_Flag'] == 1)]
extended_leave = fridays.merge(mondays, left_on=['name', 'Next_Monday'], right_on=['name', 'Date'], how='inner')

# Get unique employee names who took extended leave
suspects = extended_leave['name'].unique()

import matplotlib.dates as mdates
import matplotlib.pyplot as plt
import seaborn as sns

# Step 1: Group by date and calculate average presence
daily_trend = df_long.groupby('Date')['Present'].mean().reset_index()

# Step 2: Identify min and max attendance dates
max_day = daily_trend[daily_trend['Present'] == daily_trend['Present'].max()]
min_day = daily_trend[daily_trend['Present'] == daily_trend['Present'].min()]

# Step 3: Plot with enhancements
plt.figure(figsize=(14, 5))
sns.lineplot(data=daily_trend, x='Date', y='Present')

# Highlight max and min points
plt.scatter(max_day['Date'], max_day['Present'], color='green', label='Max Attendance')
plt.text(max_day['Date'].values[0], max_day['Present'].values[0] + 0.03,
         f" {max_day['Present'].values[0]:.2f}", color='green')

plt.scatter(min_day['Date'], min_day['Present'], color='red', label='Min Attendance')
plt.text(min_day['Date'].values[0], min_day['Present'].values[0] - 0.05,
         f" {min_day['Present'].values[0]:.2f}", color='red')

# Improve axis formatting
plt.gca().xaxis.set_major_locator(mdates.DayLocator(interval=1))  # Show all dates
plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %d'))  # Format as "Mar 05"
plt.xticks(rotation=45)

plt.ylim(0, 1)  # Y-axis from 0 to 100%
plt.title("Daily Team Attendance Trend")
plt.xlabel("Date")
plt.ylabel("Avg. Presence Rate")
plt.grid(True, linestyle='--', alpha=0.5)
plt.legend()
plt.tight_layout()
plt.show()
